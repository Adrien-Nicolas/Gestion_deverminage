1262  private function _parse_css($str)
1263  {
1264  $str = trim($str);
1265 
1266  // Destroy comments and remove HTML comments
1267  $css = preg_replace([
1268  "'/\*.*?\*/'si",
1269  "/^<!--/",
1270  "/-->$/"
1271  ], "", $str);
1272 
1273  // FIXME: handle '{' within strings, e.g. [attr="string {}"]
1274 
1275  // Something more legible:
1276  $re =
1277  "/\s* # Skip leading whitespace \n" .
1278  "( @([^\s{]+)\s*([^{;]*) (?:;|({)) )? # Match @rules followed by ';' or '{' \n" .
1279  "(?(1) # Only parse sub-sections if we're in an @rule... \n" .
1280  " (?(4) # ...and if there was a leading '{' \n" .
1281  " \s*( (?:(?>[^{}]+) ({)? # Parse rulesets and individual @page rules \n" .
1282  " (?(6) (?>[^}]*) }) \s*)+? \n" .
1283  " ) \n" .
1284  " }) # Balancing '}' \n" .
1285  "| # Branch to match regular rules (not preceded by '@') \n" .
1286  "([^{]*{[^}]*})) # Parse normal rulesets \n" .
1287  "/xs";
1288 
1289  if (preg_match_all($re, $css, $matches, PREG_SET_ORDER) === false) {
1290  // An error occurred
1291  throw new Exception("Error parsing css file: preg_match_all() failed.");
1292  }
1293 
1294  // After matching, the array indices are set as follows:
1295  //
1296  // [0] => complete text of match
1297  // [1] => contains '@import ...;' or '@media {' if applicable
1298  // [2] => text following @ for cases where [1] is set
1299  // [3] => media types or full text following '@import ...;'
1300  // [4] => '{', if present
1301  // [5] => rulesets within media rules
1302  // [6] => '{', within media rules
1303  // [7] => individual rules, outside of media rules
1304  //
1305 
1306  $media_query_regex = "/(?:((only|not)?\s*(" . implode("|", self::$VALID_MEDIA_TYPES) . "))|(\s*\‍(\s*((?:(min|max)-)?([\w\-]+))\s*(?:\:\s*(.*?)\s*)?\‍)))/isx";
1307 
1308  //Helpers::pre_r($matches);
1309  foreach ($matches as $match) {
1310  $match[2] = trim($match[2]);
1311 
1312  if ($match[2] !== "") {
1313  // Handle @rules
1314  switch ($match[2]) {
1315 
1316  case "import":
1317  $this->_parse_import($match[3]);
1318  break;
1319 
1320  case "media":
1321  $acceptedmedia = self::$ACCEPTED_GENERIC_MEDIA_TYPES;
1322  $acceptedmedia[] = $this->_dompdf->getOptions()->getDefaultMediaType();
1323 
1324  $media_queries = preg_split("/\s*,\s*/", mb_strtolower(trim($match[3])));
1325  foreach ($media_queries as $media_query) {
1326  if (in_array($media_query, $acceptedmedia)) {
1327  //if we have a media type match go ahead and parse the stylesheet
1328  $this->_parse_sections($match[5]);
1329  break;
1330  } elseif (!in_array($media_query, self::$VALID_MEDIA_TYPES)) {
1331  // otherwise conditionally parse the stylesheet assuming there are parseable media queries
1332  if (preg_match_all($media_query_regex, $media_query, $media_query_matches, PREG_SET_ORDER) !== false) {
1333  $mq = [];
1334  foreach ($media_query_matches as $media_query_match) {
1335  if (empty($media_query_match[1]) === false) {
1336  $media_query_feature = strtolower($media_query_match[3]);
1337  $media_query_value = strtolower($media_query_match[2]);
1338  $mq[] = [$media_query_feature, $media_query_value];
1339  } else if (empty($media_query_match[4]) === false) {
1340  $media_query_feature = strtolower($media_query_match[5]);
1341  $media_query_value = (array_key_exists(8, $media_query_match) ? strtolower($media_query_match[8]) : null);
1342  $mq[] = [$media_query_feature, $media_query_value];
1343  }
1344  }
1345  $this->_parse_sections($match[5], $mq);
1346  break;
1347  }
1348  }
1349  }
1350  break;
1351 
1352  case "page":
1353  //This handles @page to be applied to page oriented media
1354  //Note: This has a reduced syntax:
1355  //@page { margin:1cm; color:blue; }
1356  //Not a sequence of styles like a full.css, but only the properties
1357  //of a single style, which is applied to the very first "root" frame before
1358  //processing other styles of the frame.
1359  //Working properties:
1360  // margin (for margin around edge of paper)
1361  // font-family (default font of pages)
1362  // color (default text color of pages)
1363  //Non working properties:
1364  // border
1365  // padding
1366  // background-color
1367  //Todo:Reason is unknown
1368  //Other properties (like further font or border attributes) not tested.
1369  //If a border or background color around each paper sheet is desired,
1370  //assign it to the <body> tag, possibly only for the css of the correct media type.
1371 
1372  // If the page has a name, skip the style.
1373  $page_selector = trim($match[3]);
1374 
1375  $key = null;
1376  switch ($page_selector) {
1377  case "":
1378  $key = "base";
1379  break;
1380 
1381  case ":left":
1382  case ":right":
1383  case ":odd":
1384  case ":even":
1386  case ":first":
1387  $key = $page_selector;
1388  break;
1389 
1390  default:
1391  break 2;
1392  }
1393 
1394  // Store the style for later...
1395  if (empty($this->_page_styles[$key])) {
1396  $this->_page_styles[$key] = $this->_parse_properties($match[5]);
1397  } else {
1398  $this->_page_styles[$key]->merge($this->_parse_properties($match[5]));
1399  }
1400  break;
1401 
1402  case "font-face":
1403  $this->_parse_font_face($match[5]);
1404  break;
1405 
1406  default:
1407  // ignore everything else
1408  break;
1409  }
1410 
1411  continue;
1412  }
1413 
1414  if ($match[7] !== "") {
1415  $this->_parse_sections($match[7]);
1416  }
1417  }
1418  }
