4844  if (count($bytes) === 0) { // get starting octect

4849  $bytes[] = ($c - 0xC0) << 0x06;

4856  $numbytes = 4;
4857  } else {
4858  // use replacement character for other invalid sequences
4859  $unicode[] = 0xFFFD;
4860  $bytes = [];
4861  $numbytes = 1;
4862  }
4863  } elseif (($c >> 0x06) === 0x02) { // bytes 2, 3 and 4 must start with 0x02 = 10 BIN
4864  $bytes[] = $c - 0x80;
