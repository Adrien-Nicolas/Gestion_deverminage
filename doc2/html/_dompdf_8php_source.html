617  $stylesheets = $xpath->query("//*[name() = 'link' or name() = 'style']");
618

622  // load <link rel="STYLESHEET" ... /> tags
623  case "link":

625  mb_strtolower($tag->getAttribute("type")) === "text/css"
626  ) {
627  //Check if the css file is for an accepted media type
628  //media not given then always valid

634  $accept = true;
635  break;
636  }
637  }

641  //Skip this css file.
642  break;
